1 /**
  2  * JSBitStream is a Javascript class to read and write bit level data into and out of a stream.
  3  * It was created to preserve as many bits and bytes during network communication as possible without sacrificing
  4  * speed (ie. through compression). It was intended to prepare network packets for multiplayer HTML5 games.
  5  *
  6  * @class
  7  * @name JSBitStream
  8  * @namespace
  9  */
 10 var JSBitStream = function () {
 11     'use strict';
 12 };
 13 
 14 /**
 15  * The string buffer packing 16 bits into each Unicode character - every character represents two 8 bit values
 16  * @private
 17  * @type {String}
 18  */
 19 JSBitStream.prototype.data = "";
 20 
 21 /**
 22  * The number of bits used in the first character
 23  * @private
 24  * @type {Number}
 25  */
 26 JSBitStream.prototype.bitOffset = 0;
 27 
 28 /**
 29  * The number of used bits (0-15) in the final incomplete 16 bit character (0 if all 16 bits are used)
 30  * @private
 31  * @type {Number}
 32  */
 33 JSBitStream.prototype.lastCharBits = 0;
 34 
 35 /**
 36  * Reads an arbitrary string from the stream.
 37  * @public
 38  * @return {String} The string read.
 39  */
 40 JSBitStream.prototype.readString = function () {
 41     'use strict';
 42 
 43     var typeId = this.readU4(),
 44         l = this.readU16(),
 45         txt = "",
 46         c;
 47 
 48     while (l--) {
 49         switch (typeId) {
 50         case 5:  // numeric 4 bit
 51             c = this.readBits(4).data.charCodeAt(0) >> 12;
 52             if (c === 4) {
 53                 c = 32;
 54             } else {
 55                 c += 43;
 56             }
 57             break;
 58         case 4:  // lowercase alpha 5 bit
 59             c = this.readBits(5).data.charCodeAt(0) >> 11;
 60             if (c === 26) {         // space
 61                 c = 32;
 62             } else if (c === 27) {  // |
 63                 c = 124;
 64             } else if (c === 28) {  // '
 65                 c = 39;
 66             } else if (c === 29) {  // -
 67                 c = 45;
 68             } else if (c === 30) {  // .
 69                 c = 46;
 70             } else if (c === 31) {  // ,
 71                 c = 44;
 72             } else {
 73                 c += 97;           // a-z
 74             }
 75             break;
 76         case 3:  // alphanumeric 6 bit
 77             c = this.readBits(6).data.charCodeAt(0) >> 10;
 78             if (c === 62) {         // ,
 79                 c = 44;
 80             } else if (c === 63) {  // space
 81                 c = 32;
 82             } else {
 83                 c += 48;
 84                 if (c >= 58) { c += 7; }
 85                 if (c >= 91) { c += 6; }
 86             }
 87             break;
 88         case 2:  // low ascii 7 bit
 89             c = this.readBits(7).data.charCodeAt(0) >> 9;
 90             break;
 91         case 1:  // ascii 8 bit
 92             c = this.readBits(8).data.charCodeAt(0) >> 8;
 93             break;
 94         default: // unicode
 95             c = this.readBits(16).data.charCodeAt(0);
 96             break;
 97         }
 98 
 99         txt += String.fromCharCode(c);
100     }
101 
102     return txt;
103 };

105 /**
106  * Writes an arbitrary string into the bitstream.
107  * @public
108  * @param val (string) The string value to be written.
109  * @param lowerCase (boolean) Shows whether the string can be represented in a lower case format.
110  *                            Defaults to false. A full lowercase string takes less bits in the bitstream.
111  * @return {String} The string that was inserted into the bitstream.
112  */
113 JSBitStream.prototype.writeString = function (val, lowerCase) {
114     'use strict';
115 
116     if (lowerCase === undefined) {
117         lowerCase = false;
118     }
119 
120     if (lowerCase) {
121         val = val.toLowerCase();
122     }
123 
124     var numericOnly = true,
125         lowerCaseCharactersOnly = true,
126         alphanumericOnly = true,
127         lowASCIIOnly = true,
128         asciiOnly = true,
129         typeId,
130         c,
131         t;
132 
133     for (t = 0; t < val.length; t++) {
134         c = val.charCodeAt(t);
135         if ((c < 43 || c > 57 || c === 47) && c !== 32) { numericOnly = false; } // includes: +,-.space
136         if ((c < 97 || c > 122) && c !== 32 && c !== 124 && c !== 39 && c !== 46 && c !== 44) { lowerCaseCharactersOnly = false; } // includes '|-., and space
137         if ((c < 48 || c > 57) && (c < 65 || c > 90) && (c < 97 || c > 122) && (c !== 32 && c !== 39)) { alphanumericOnly = false; } // includes , and space
138         if (c > 127) { lowASCIIOnly = false; }
139         if (c > 255) { asciiOnly = false; }
140     }
141 
142     typeId = (numericOnly ? 5 : (lowerCaseCharactersOnly ? 4 : (alphanumericOnly ? 3 : (lowASCIIOnly ? 2 : (asciiOnly ? 1 : 0))))) & 0x000F;
143 
144     this.writeU4(typeId);
145     this.writeU16(val.length);
146 
147     for (t = 0; t < val.length; t++) {
148         c = val.charCodeAt(t);
149         switch (typeId) {
150         case 5:     // 5 - numeric only (4 bits) (0-15) - includes: +,-.space
151             if (c === 32) {
152                 c = 4; // space instead of /
153             } else {
154                 c -= 43;
155             }
156             this.writeBits(String.fromCharCode((c << 12) & 0xF000), 4);
157             break;
158         case 4:     // 4 - lowercase alpha only (5 bits) (0-31) - includes space|'-.,
159             if (c === 32) {         // space
160                 c = 26;
161             } else if (c === 124) {  // |
162                 c = 27;
163             } else if (c === 39) {  // '
164                 c = 28;
165             } else if (c === 45) {  // -
166                 c = 29;
167             } else if (c === 46) {  // .
168                 c = 30;
169             } else if (c === 44) {  // ,
170                 c = 31;
171             } else {
172                 c -= 97;          // a-z
173             }
174             this.writeBits(String.fromCharCode((c << 11) & 0xF800), 5);
175             break;
176         case 3:     // 3 - alphanumeric only (6 bits) (0-63) - includes , and space
177             if (c === 44) {         // ,
178                 c = 62;
179             } else if (c === 32) {  // space
180                 c = 63;
181             } else {
182                 c -= 48;
183                 if (c >= 17) { c -= 7; }
184                 if (c >= 42) { c -= 6; }
185             }
186             this.writeBits(String.fromCharCode((c << 10) & 0xFC00), 6);
187             break;
188         case 2:     // 2 - low ascii only (7 bit) (0-127)
189             this.writeBits(String.fromCharCode((c << 9) & 0xFE00), 7);
190             break;
191         case 1:     // 1 - ascii only (8 bits)
192             this.writeBits(String.fromCharCode((c << 8) & 0xFF00), 8);
193             break;
194         default:    // 0 - unicode (16 bits)
195             this.writeBits(String.fromCharCode(c & 0xFFFF), 16);
196             break;
197         }
198     }
199 
200     return val;
201 };

203 /**
204  * Reads a compressed integer from the bitstream. Useful for numbers where small (<256) and
205  * large (>65536 or >4294967296) values fluctuate.
206  * @public
207  * @return {Number|String} The integer read from the bitstream. For small numbers, this is a number, for large numbers
208  *                         this is a string.
209  */
210 JSBitStream.prototype.readInt = function () {
211     'use strict';
212 
213     if (this.readFlag()) { // 8 bit?
214         return this.readU8();
215     }
216     if (this.readFlag()) { // 16 bit ?
217         return this.readU16();
218     }
219     if (this.readFlag()) { // 32 bit ?
220         return this.readU32();
221     }
222     // large value
223     return this.readString();
224 };
225 
226 /**
227  * Writes a compressed integer into the bitstream. Use for numbers where small (<256)
228  * and large (>65536 or >4294967296) values fluctuate.
229  * @public
230  * @param val The number to be written into the bitstream. For extremely large values, this is treated as a string.
231  * @return {Number|String} The value written into the bitstream.
232  */
233 JSBitStream.prototype.writeInt = function (val) {
234     'use strict';
235 
236     val *= 1;
237     if (this.writeFlag(val < Math.pow(2, 8))) { // 8 bit
238         this.writeU8(val);
239     } else if (this.writeFlag(val < Math.pow(2, 16))) { // 16 bit
240         this.writeU16(val);
241     } else if (this.writeFlag(val < Math.pow(2, 32))) { // 32 bit
242         this.writeU32(val);
243     } else { // large value
244         this.writeString(val.toString(), false);
245     }
246 
247     return val;
248 };

250 /**
251  * Reads a relative float (0-1) value from the bitstream with 8 bit precision.
252  * @public
253  * @return {Number} The number read from the bitstream. Note that this number will have a difference of up to 0.008.
254  */
255 JSBitStream.prototype.readFloat = function () {
256     'use strict';
257 
258     return (this.readU8() & 0xFF) / 255;
259 };
260 
261 /**
262  * Writes a relative float (0-1) value into the bitstream with 8 bit precision.
263  * @public
264  * @param val_float The value to be written into the bitstream. The value must be between 0 and 1 (inclusive).
265  * @return {Number} The number written into the bitstream (not the number passed as the original argument).
266  */
267 JSBitStream.prototype.writeFloat = function (val_float) {
268     'use strict';
269 
270     val_float = (val_float * 255) & 0xFF;
271     this.writeU8(val_float);
272     return val_float;
273 };

275 /**
276  * Reads a 32 bit value from the bitstream.
277  * @public
278  * @return {Number} The 32 bit number read from the bitstream.
279  */
280 JSBitStream.prototype.readU32 = function () {
281     'use strict';
282 
283     var readStr = this.readBits(32).data.toString();
284     return (readStr.charCodeAt(0) & 0xFFFF) * 0x10000 + (readStr.charCodeAt(1) & 0xFFFF);
285 };

396 /**
397  * Reads an arbitrary amount of bits and returns it as a JSBitStream object. This always reads from bitOffset. The
398  * final portion of the stream that was read is then shifted to have a 0 offset before converting to a type.
399  * @private
400  * @param count The number of bits to be read from the bitstream.
401  * @return {JSBitStream} The JSBitStream object representing the bits read.
402  */
403 JSBitStream.prototype.readBits = function (count) {
404     'use strict';
405 
406     if (this.size() < count) {
407         count = this.size();
408     }
409 
410     // TODO: does this work with node?
411     var readStream = new JSBitStream(),
412         toReadCount,
413         prevBitOffset = this.bitOffset,
414         originalCount = count,
415         firstChar = "",
416         readMsg,
417         r;
418 
419     if (this.debug) {
420         this.log(0, "readBits reading " + originalCount + " bit(s) with a bit offset of " + prevBitOffset + " within the first character");
421     }
422 
423     if (this.bitOffset > 0) {
424         toReadCount = Math.min(16 - this.bitOffset, count);
425 
426         // copy the first character using only valid bits
427         firstChar = ((this.data.charCodeAt(0) << this.bitOffset) & ((Math.pow(2, toReadCount) - 1) << (16 - toReadCount)));
428         if (this.debug) {
429             this.log(0, "readBits initial offset read: " + this.left_pad(firstChar.toString(2), toReadCount, "0").substr(0, toReadCount));
430         }
431         readStream.writeBits(String.fromCharCode(firstChar), toReadCount);
432 
433         this.bitOffset = (this.bitOffset + toReadCount) % 16;
434         if (this.bitOffset === 0) {
435             this.data = this.data.substr(1);
436         }
437 
438         count -= toReadCount;
439     }
440 
441     while (count > 0) {
442         toReadCount = Math.min(16, count);
443 
444         readStream.writeBits(String.fromCharCode((this.data.charCodeAt(0) & ((Math.pow(2, toReadCount) - 1) << (16 - toReadCount)))), toReadCount);
445 
446         if (toReadCount < 16) {
447             // the read ends in this character with a non-zero bitOffset
448             this.bitOffset = toReadCount;
449         } else {
450             // the read ends with the last bit or goes on in the next character
451             this.data = this.data.substr(1);
452         }
453 
454         count -= toReadCount;
455     }
456 
457     if (this.debug && originalCount > 0) {
458         readMsg = "Read '";
459         if (firstChar.length > 0) {
460             readMsg += this.left_pad(firstChar.charCodeAt(0).toString(2), 16 - prevBitOffset, "0") + " ";
461         }
462         for (r = 0; r < readStream.data.length; r++) {
463             readMsg += this.left_pad(readStream.data.charCodeAt(r).toString(2), 16, "0").substr(0, Math.min(originalCount - r * 16, 16)) + " ";
464         }
465         readMsg = readMsg.trim();
466         readMsg += "'";
467         this.log(0, readMsg);
468     }
469 
470     // clean up if no data is left in the stream
471     if (this.size === 0) {
472         this.data = "";
473         this.bitOffset = 0;
474         this.lastCharBits = 0;
475     }
476 
477     if (this.debug) {
478         this.log(0, this.serialize(false));
479     }
480 
481     return readStream;
482 };

484 /**
485  * Writes an arbitrary amount of bits into the stream.
486  * @private
487  * @param writeBuffer {String} A bit packed string containing the value to be written.
488  * @param bitCount {Number} The number of bits to write. Must be equal or less than 16 * the number of characters in
489  *                          writeBuffer.
490  */
491 JSBitStream.prototype.writeBits = function (writeBuffer, bitCount) {
492     'use strict';
493 
494     bitCount = Math.min(bitCount, writeBuffer.length * 16);
495 
496     var writeBufferPointer = 0,
497         bitsToRead,
498         writeValue,
499         targetOffset,
500         targetValue,
501         firstBitsToWrite,
502         firstValue,
503         nextBitsToWrite,
504         nextValue;
505 
506     while (writeBufferPointer < bitCount) {
507         // get at most 16 bits and add them as a new character
508         // read the buffer to write (writeBuffer) up to the end of the current character or to the end of the string (bitCount)
509         bitsToRead = Math.min(Math.min(16, (bitCount - writeBufferPointer)), 16 - (writeBufferPointer % 16));
510         // this is the value we need to write - masked out from the source character and shifted to the right
511         writeValue = (writeBuffer.charCodeAt(writeBufferPointer / 16 >> 0) >> (16 - bitsToRead)) & (Math.pow(2, bitsToRead) - 1);
512         // this is the character we're writing to
513         targetOffset = Math.max(this.data.length - 1 + (this.lastCharBits === 0 ? 1 : 0), 0);
514 
515         if (this.data.length - 1 < targetOffset) {
516             this.data += "\u0000";
517         }
518 
519         firstBitsToWrite = Math.min(16 - this.lastCharBits, bitsToRead);
520         firstValue = writeValue >> (bitsToRead - firstBitsToWrite);
521         nextBitsToWrite = bitsToRead - firstBitsToWrite;
522         nextValue = writeValue & (Math.pow(2, nextBitsToWrite) - 1);
523 
524         // mask out any bits following the last write from the target value
525         // add the new value shifted to the target position
526         // and copy it into the target stream's last character
527         targetValue = this.data.charCodeAt(targetOffset) & 0xFFFF;
528         targetValue &= ((Math.pow(2, this.lastCharBits) - 1) << (16 - this.lastCharBits));
529         targetValue |= firstValue << (16 - (this.lastCharBits + firstBitsToWrite));
530         this.data = this.data.substr(0, targetOffset) + String.fromCharCode(targetValue & 0xFFFF);
531 
532         if (nextBitsToWrite > 0) {
533             // the value to be written has overflown the target character
534             // add a new zero character at the end of the stream
535             // and shift the remaining bits that need to be written all the way to the left
536             // before writing it into the stream's new character
537             this.data += "\u0000";
538             targetValue = nextValue << (16 - nextBitsToWrite);
539             this.data = this.data.substr(0, targetOffset + 1) + String.fromCharCode(targetValue & 0xFFFF);
540         }
541 
542         if (this.debug) {
543             if (bitCount > 32) {
544                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(0) & 0x00FF).toString(16), 2, "0") + " " +
545                     this.left_pad(((writeBuffer.charCodeAt(1) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(1) & 0x00FF).toString(16), 2, "0") + " " +
546                     this.left_pad(((writeBuffer.charCodeAt(2) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(2) & 0x00FF).toString(16), 2, "0") + " " +
547                     this.left_pad(((writeBuffer.charCodeAt(3) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(3) & 0x00FF).toString(16), 2, "0"));
548             } else if (bitCount > 16) {
549                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(0) & 0x00FF).toString(16), 2, "0") + " " +
550                     this.left_pad(((writeBuffer.charCodeAt(1) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(1) & 0x00FF).toString(16), 2, "0"));
551             } else if (bitCount > 8) {
552                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(0) & 0x00FF).toString(16), 2, "0"));
553             } else if (bitCount > 1) {
554                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0xFF00) >> 8).toString(16), 2, "0"));
555             } else {
556                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0x8000) >> 15).toString(16), 1, "0"));
557             }
558             this.log(0, "writeBits preparing to write " + bitsToRead + " of " + bitCount + " bit(s)" + (writeBufferPointer > 0 ? " at read offset " + writeBufferPointer : ""));
559             this.log(0, "writeBits writing " + bitsToRead + " bit(s) with a value of '" + this.left_pad(writeBuffer.charCodeAt((writeBufferPointer / 16) >> 0).toString(2), 16, "0").substr(0, bitsToRead) + "' from " + targetOffset + "." + (this.lastCharBits) + " to " + ((nextBitsToWrite > 0) ? targetOffset + 1 : targetOffset) + "." + ((this.lastCharBits + bitsToRead) % 16));
560         }
561 
562         this.lastCharBits = (this.lastCharBits + bitsToRead) % 16;
563         writeBufferPointer += bitsToRead;
564 
565         if (this.debug) {
566             this.lastBitsAdded = bitsToRead;
567             this.log(0, this.serialize(true));
568         }
569     }
570 };

572 /**
573  * Returns the number of bits within the stream.
574  * @return {Number} teh number of bits within the stream.
575  */
576 JSBitStream.prototype.size = function () {
577     'use strict';
578 
579     return (this.data.length * 16) - this.bitOffset - ((16 - this.lastCharBits) % 16);
