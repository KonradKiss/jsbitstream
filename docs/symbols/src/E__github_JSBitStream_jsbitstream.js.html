1 /**
  2  * JSBitStream is a Javascript class to read and write bit level data into and out of a stream.
  3  * It was created to preserve as many bits and bytes during network communication as possible without sacrificing
  4  * speed (ie. through compression). It was intended to prepare network packets for multiplayer HTML5 games.
  5  *
  6  * @class
  7  * @name JSBitStream
  8  * @namespace
  9  */
 10 var JSBitStream = function () {
 11     'use strict';
 12 };
 13 
 14 /**
 15  * The string buffer packing 16 bits into each Unicode character - every character represents two 8 bit values
 16  * @type {String}
 17  */
 18 JSBitStream.prototype.data = "";
 19 
 20 /**
 21  * The number of bits used in the first character
 22  * @type {Number}
 23  */
 24 JSBitStream.prototype.bitOffset = 0;
 25 
 26 /**
 27  * The number of used bits (0-15) in the final incomplete 16 bit character (0 if all 16 bits are used)
 28  * @type {Number}
 29  */
 30 JSBitStream.prototype.lastCharBits = 0;

 33 /**
 34  * Reads an arbitrary string from the stream.
 35  * @return {String} The string read.
 36  */
 37 JSBitStream.prototype.readString = function () {
 38     'use strict';
 39 
 40     var typeId = this.readU4(),
 41         l = this.readU16(),
 42         txt = "",
 43         c;
 44 
 45     while (l--) {
 46         switch (typeId) {
 47         case 5:  // numeric 4 bit
 48             c = this.readBits(4).data.charCodeAt(0) >> 12;
 49             if (c === 4) {
 50                 c = 32;
 51             } else {
 52                 c += 43;
 53             }
 54             break;
 55         case 4:  // lowercase alpha 5 bit
 56             c = this.readBits(5).data.charCodeAt(0) >> 11;
 57             if (c === 26) {         // space
 58                 c = 32;
 59             } else if (c === 27) {  // |
 60                 c = 124;
 61             } else if (c === 28) {  // '
 62                 c = 39;
 63             } else if (c === 29) {  // -
 64                 c = 45;
 65             } else if (c === 30) {  // .
 66                 c = 46;
 67             } else if (c === 31) {  // ,
 68                 c = 44;
 69             } else {
 70                 c += 97;           // a-z
 71             }
 72             break;
 73         case 3:  // alphanumeric 6 bit
 74             c = this.readBits(6).data.charCodeAt(0) >> 10;
 75             if (c === 62) {         // ,
 76                 c = 44;
 77             } else if (c === 63) {  // space
 78                 c = 32;
 79             } else {
 80                 c += 48;
 81                 if (c >= 58) { c += 7; }
 82                 if (c >= 91) { c += 6; }
 83             }
 84             break;
 85         case 2:  // low ascii 7 bit
 86             c = this.readBits(7).data.charCodeAt(0) >> 9;
 87             break;
 88         case 1:  // ascii 8 bit
 89             c = this.readBits(8).data.charCodeAt(0) >> 8;
 90             break;
 91         default: // unicode
 92             c = this.readBits(16).data.charCodeAt(0);
 93             break;
 94         }
 95 
 96         txt += String.fromCharCode(c);
 97     }
 98 
 99     return txt;
100 };

102 /**
103  * Writes an arbitrary string into the bitstream.
104  * @param val (string) The string value to be written.
105  * @param lowerCase (boolean) Shows whether the string can be represented in a lower case format.
106  *                            Defaults to false. A full lowercase string takes less bits in the bitstream.
107  * @return {String} The string that was inserted into the bitstream.
108  */
109 JSBitStream.prototype.writeString = function (val, lowerCase) {
110     'use strict';
111 
112     if (lowerCase === undefined) {
113         lowerCase = false;
114     }
115 
116     if (lowerCase) {
117         val = val.toLowerCase();
118     }
119 
120     var numericOnly = true,
121         lowerCaseCharactersOnly = true,
122         alphanumericOnly = true,
123         lowASCIIOnly = true,
124         asciiOnly = true,
125         typeId,
126         c,
127         t;
128 
129     for (t = 0; t < val.length; t++) {
130         c = val.charCodeAt(t);
131         if ((c < 43 || c > 57 || c === 47) && c !== 32) { numericOnly = false; } // includes: +,-.space
132         if ((c < 97 || c > 122) && c !== 32 && c !== 124 && c !== 39 && c !== 46 && c !== 44) { lowerCaseCharactersOnly = false; } // includes '|-., and space
133         if ((c < 48 || c > 57) && (c < 65 || c > 90) && (c < 97 || c > 122) && (c !== 32 && c !== 39)) { alphanumericOnly = false; } // includes , and space
134         if (c > 127) { lowASCIIOnly = false; }
135         if (c > 255) { asciiOnly = false; }
136     }
137 
138     typeId = (numericOnly ? 5 : (lowerCaseCharactersOnly ? 4 : (alphanumericOnly ? 3 : (lowASCIIOnly ? 2 : (asciiOnly ? 1 : 0))))) & 0x000F;
139 
140     this.writeU4(typeId);
141     this.writeU16(val.length);
142 
143     for (t = 0; t < val.length; t++) {
144         c = val.charCodeAt(t);
145         switch (typeId) {
146         case 5:     // 5 - numeric only (4 bits) (0-15) - includes: +,-.space
147             if (c === 32) {
148                 c = 4; // space instead of /
149             } else {
150                 c -= 43;
151             }
152             this.writeBits(String.fromCharCode((c << 12) & 0xF000), 4);
153             break;
154         case 4:     // 4 - lowercase alpha only (5 bits) (0-31) - includes space|'-.,
155             if (c === 32) {         // space
156                 c = 26;
157             } else if (c === 124) {  // |
158                 c = 27;
159             } else if (c === 39) {  // '
160                 c = 28;
161             } else if (c === 45) {  // -
162                 c = 29;
163             } else if (c === 46) {  // .
164                 c = 30;
165             } else if (c === 44) {  // ,
166                 c = 31;
167             } else {
168                 c -= 97;          // a-z
169             }
170             this.writeBits(String.fromCharCode((c << 11) & 0xF800), 5);
171             break;
172         case 3:     // 3 - alphanumeric only (6 bits) (0-63) - includes , and space
173             if (c === 44) {         // ,
174                 c = 62;
175             } else if (c === 32) {  // space
176                 c = 63;
177             } else {
178                 c -= 48;
179                 if (c >= 17) { c -= 7; }
180                 if (c >= 42) { c -= 6; }
181             }
182             this.writeBits(String.fromCharCode((c << 10) & 0xFC00), 6);
183             break;
184         case 2:     // 2 - low ascii only (7 bit) (0-127)
185             this.writeBits(String.fromCharCode((c << 9) & 0xFE00), 7);
186             break;
187         case 1:     // 1 - ascii only (8 bits)
188             this.writeBits(String.fromCharCode((c << 8) & 0xFF00), 8);
189             break;
190         default:    // 0 - unicode (16 bits)
191             this.writeBits(String.fromCharCode(c & 0xFFFF), 16);
192             break;
193         }
194     }
195 
196     return val;
197 };

199 /**
200  * Reads a compressed integer from the bitstream. Useful for numbers where small (<256) and
201  * large (>65536 or >4294967296) values fluctuate.
202  * @return {Number|String} The integer read from the bitstream. For small numbers, this is a number, for large numbers
203  *                         this is a string.
204  */
205 JSBitStream.prototype.readInt = function () {
206     'use strict';
207 
208     if (this.readFlag()) { // 8 bit?
209         return this.readU8();
210     }
211     if (this.readFlag()) { // 16 bit ?
212         return this.readU16();
213     }
214     if (this.readFlag()) { // 32 bit ?
215         return this.readU32();
216     }
217     // large value
218     return this.readString();
219 };
220 
221 /**
222  * Writes a compressed integer into the bitstream. Use for numbers where small (<256)
223  * and large (>65536 or >4294967296) values fluctuate.
224  * @param val The number to be written into the bitstream. For extremely large values, this is treated as a string.
225  * @return {Number|String} The value written into the bitstream.
226  */
227 JSBitStream.prototype.writeInt = function (val) {
228     'use strict';
229 
230     val *= 1;
231     if (this.writeFlag(val < Math.pow(2, 8))) { // 8 bit
232         this.writeU8(val);
233     } else if (this.writeFlag(val < Math.pow(2, 16))) { // 16 bit
234         this.writeU16(val);
235     } else if (this.writeFlag(val < Math.pow(2, 32))) { // 32 bit
236         this.writeU32(val);
237     } else { // large value
238         this.writeString(val.toString(), false);
239     }
240 
241     return val;
242 };

244 /**
245  * Reads a relative float (0-1) value from the bitstream with 8 bit precision.
246  * @return {Number} The number read from the bitstream. Note that this number will have a difference of up to 0.008.
247  */
248 JSBitStream.prototype.readFloat = function () {
249     'use strict';
250 
251     return (this.readU8() & 0xFF) / 255;
252 };

254 /**
255  * Writes a relative float (0-1) value into the bitstream with 8 bit precision.
256  * @param val_float The value to be written into the bitstream. The value must be between 0 and 1 (inclusive).
257  * @return {Number} The number written into the bitstream (not the number passed as the original argument).
258  */
259 JSBitStream.prototype.writeFloat = function (val_float) {
260     'use strict';
261 
262     val_float = (val_float * 255) & 0xFF;
263     this.writeU8(val_float);
264     return val_float;
265 };

267 /**
268  * Reads a 32 bit value from the bitstream.
269  * @return {Number} The 32 bit number read from the bitstream.
270  */
271 JSBitStream.prototype.readU32 = function () {
272     'use strict';
273 
274     var readStr = this.readBits(32).data.toString();
275     return (readStr.charCodeAt(0) & 0xFFFF) * 0x10000 + (readStr.charCodeAt(1) & 0xFFFF);

378 /**
379  * Reads an arbitrary amount of bits and returns it as a JSBitStream object. This always reads from bitOffset. The
380  * final portion of the stream that was read is then shifted to have a 0 offset before converting to a type.
381  * @param count The number of bits to be read from the bitstream.
382  * @return {JSBitStream} The JSBitStream object representing the bits read.
383  */
384 JSBitStream.prototype.readBits = function (count) {
385     'use strict';
386 
387     if (this.size() < count) {
388         count = this.size();
389     }
390 
391     // TODO: does this work with node?
392     var readStream = new JSBitStream(),
393         toReadCount,
394         prevBitOffset = this.bitOffset,
395         originalCount = count,
396         firstChar = "",
397         readMsg,
398         r;
399 
400     if (this.debug) {
401         this.log(0, "readBits reading " + originalCount + " bit(s) with a bit offset of " + prevBitOffset + " within the first character");
402     }
403 
404     if (this.bitOffset > 0) {
405         toReadCount = Math.min(16 - this.bitOffset, count);
406 
407         // copy the first character using only valid bits
408         firstChar = ((this.data.charCodeAt(0) << this.bitOffset) & ((Math.pow(2, toReadCount) - 1) << (16 - toReadCount)));
409         if (this.debug) {
410             this.log(0, "readBits initial offset read: " + this.left_pad(firstChar.toString(2), toReadCount, "0").substr(0, toReadCount));
411         }
412         readStream.writeBits(String.fromCharCode(firstChar), toReadCount);
413 
414         this.bitOffset = (this.bitOffset + toReadCount) % 16;
415         if (this.bitOffset === 0) {
416             this.data = this.data.substr(1);
417         }
418 
419         count -= toReadCount;
420     }
421 
422     while (count > 0) {
423         toReadCount = Math.min(16, count);
424 
425         readStream.writeBits(String.fromCharCode((this.data.charCodeAt(0) & ((Math.pow(2, toReadCount) - 1) << (16 - toReadCount)))), toReadCount);
426 
427         if (toReadCount < 16) {
428             // the read ends in this character with a non-zero bitOffset
429             this.bitOffset = toReadCount;
430         } else {
431             // the read ends with the last bit or goes on in the next character
432             this.data = this.data.substr(1);
433         }
434 
435         count -= toReadCount;
436     }
437 
438     if (this.debug && originalCount > 0) {
439         readMsg = "Read '";
440         if (firstChar.length > 0) {
441             readMsg += this.left_pad(firstChar.charCodeAt(0).toString(2), 16 - prevBitOffset, "0") + " ";
442         }
443         for (r = 0; r < readStream.data.length; r++) {
444             readMsg += this.left_pad(readStream.data.charCodeAt(r).toString(2), 16, "0").substr(0, Math.min(originalCount - r * 16, 16)) + " ";
445         }
446         readMsg = readMsg.trim();
447         readMsg += "'";
448         this.log(0, readMsg);
449     }
450 
451     // clean up if no data is left in the stream
452     if (this.size === 0) {
453         this.data = "";
454         this.bitOffset = 0;
455         this.lastCharBits = 0;
456     }
457 
458     if (this.debug) {
459         this.log(0, this.serialize(false));
460     }
461 
462     return readStream;
463 };

465 /**
466  * Writes an arbitrary amount of bits into the stream.
467  * @param writeBuffer {String} A bit packed string containing the value to be written.
468  * @param bitCount {Number} The number of bits to write. Must be equal or less than 16 * the number of characters in
469  *                          writeBuffer.
470  */
471 JSBitStream.prototype.writeBits = function (writeBuffer, bitCount) {
472     'use strict';
473 
474     bitCount = Math.min(bitCount, writeBuffer.length * 16);
475 
476     var writeBufferPointer = 0,
477         bitsToRead,
478         writeValue,
479         targetOffset,
480         targetValue,
481         firstBitsToWrite,
482         firstValue,
483         nextBitsToWrite,
484         nextValue;
485 
486     while (writeBufferPointer < bitCount) {
487         // get at most 16 bits and add them as a new character
488         // read the buffer to write (writeBuffer) up to the end of the current character or to the end of the string (bitCount)
489         bitsToRead = Math.min(Math.min(16, (bitCount - writeBufferPointer)), 16 - (writeBufferPointer % 16));
490         // this is the value we need to write - masked out from the source character and shifted to the right
491         writeValue = (writeBuffer.charCodeAt(writeBufferPointer / 16 >> 0) >> (16 - bitsToRead)) & (Math.pow(2, bitsToRead) - 1);
492         // this is the character we're writing to
493         targetOffset = Math.max(this.data.length - 1 + (this.lastCharBits === 0 ? 1 : 0), 0);
494 
495         if (this.data.length - 1 < targetOffset) {
496             this.data += "\u0000";
497         }
498 
499         firstBitsToWrite = Math.min(16 - this.lastCharBits, bitsToRead);
500         firstValue = writeValue >> (bitsToRead - firstBitsToWrite);
501         nextBitsToWrite = bitsToRead - firstBitsToWrite;
502         nextValue = writeValue & (Math.pow(2, nextBitsToWrite) - 1);
503 
504         // mask out any bits following the last write from the target value
505         // add the new value shifted to the target position
506         // and copy it into the target stream's last character
507         targetValue = this.data.charCodeAt(targetOffset) & 0xFFFF;
508         targetValue &= ((Math.pow(2, this.lastCharBits) - 1) << (16 - this.lastCharBits));
509         targetValue |= firstValue << (16 - (this.lastCharBits + firstBitsToWrite));
510         this.data = this.data.substr(0, targetOffset) + String.fromCharCode(targetValue & 0xFFFF);
511 
512         if (nextBitsToWrite > 0) {
513             // the value to be written has overflown the target character
514             // add a new zero character at the end of the stream
515             // and shift the remaining bits that need to be written all the way to the left
516             // before writing it into the stream's new character
517             this.data += "\u0000";
518             targetValue = nextValue << (16 - nextBitsToWrite);
519             this.data = this.data.substr(0, targetOffset + 1) + String.fromCharCode(targetValue & 0xFFFF);
520         }
521 
522         if (this.debug) {
523             if (bitCount > 32) {
524                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(0) & 0x00FF).toString(16), 2, "0") + " " +
525                     this.left_pad(((writeBuffer.charCodeAt(1) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(1) & 0x00FF).toString(16), 2, "0") + " " +
526                     this.left_pad(((writeBuffer.charCodeAt(2) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(2) & 0x00FF).toString(16), 2, "0") + " " +
527                     this.left_pad(((writeBuffer.charCodeAt(3) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(3) & 0x00FF).toString(16), 2, "0"));
528             } else if (bitCount > 16) {
529                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(0) & 0x00FF).toString(16), 2, "0") + " " +
530                     this.left_pad(((writeBuffer.charCodeAt(1) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(1) & 0x00FF).toString(16), 2, "0"));
531             } else if (bitCount > 8) {
532                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0xFF00) >> 8).toString(16), 2, "0") + " " + this.left_pad((writeBuffer.charCodeAt(0) & 0x00FF).toString(16), 2, "0"));
533             } else if (bitCount > 1) {
534                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0xFF00) >> 8).toString(16), 2, "0"));
535             } else {
536                 this.log(0, this.left_pad(((writeBuffer.charCodeAt(0) & 0x8000) >> 15).toString(16), 1, "0"));
537             }
538             this.log(0, "writeBits preparing to write " + bitsToRead + " of " + bitCount + " bit(s)" + (writeBufferPointer > 0 ? " at read offset " + writeBufferPointer : ""));
539             this.log(0, "writeBits writing " + bitsToRead + " bit(s) with a value of '" + this.left_pad(writeBuffer.charCodeAt((writeBufferPointer / 16) >> 0).toString(2), 16, "0").substr(0, bitsToRead) + "' from " + targetOffset + "." + (this.lastCharBits) + " to " + ((nextBitsToWrite > 0) ? targetOffset + 1 : targetOffset) + "." + ((this.lastCharBits + bitsToRead) % 16));
540         }
541 
542         this.lastCharBits = (this.lastCharBits + bitsToRead) % 16;
543         writeBufferPointer += bitsToRead;
544 
545         if (this.debug) {
546             this.lastBitsAdded = bitsToRead;
547             this.log(0, this.serialize(true));
548         }
549     }
550 };

552 /**
553  * Returns the number of bits within the stream.
554  * @return {Number} teh number of bits within the stream.
555  */
556 JSBitStream.prototype.size = function () {
557     'use strict';
558 
559     return (this.data.length * 16) - this.bitOffset - ((16 - this.lastCharBits) % 16);

562 /**
563  * If QUnit is available it is safe to assume that JSBitStream is being unit tested so it should provide tools that
564  * provide additional info within the console:
565  */
566 if (QUnit !== undefined) {
567     'use strict';
568 
569     JSBitStream.prototype.debug = true;
570     JSBitStream.prototype.lastBitsAdded = 0;  // debug only, shows how many bits have been added to the stream during the latest write
571 
572     /**
573      * Serializes the bitstream for debugging.
574      * @param showLastAdd {Boolean} Show the last bits added. Defaults to false.
575      * @return {String} The serialized debug information.
576      */
577     JSBitStream.prototype.serialize = function (showLastAdd) {
578         if (showLastAdd === undefined) {
579             showLastAdd = false;
580         }
581 
582         if (this.data.length === 0) {
583             return "\n<  empty  >";
584         }
585 
586         var txt = "\n",
587             chr,
588             bin,
589             lo,
590             hi,
591             lo16,
592             hi16,
593             lo2,
594             hi2,
595             strBit,
596             endBit,
597             c,
598             f;
599 
600         for (c = 0; c < this.data.length; c++) {
601             chr = this.data.charCodeAt(c);
602 
603             lo = chr & 0xFF;
604             hi = chr >> 8;
605             lo16 = this.left_pad(lo.toString(16), 2, "0");
606             hi16 = this.left_pad(hi.toString(16), 2, "0");
607             lo2 = this.left_pad(lo.toString(2), 8, "0");
608             hi2 = this.left_pad(hi.toString(2), 8, "0");
609 
610             bin = hi2 + lo2;
611 
612             if (c === 0) {
613                 bin = this.left_pad("", this.bitOffset, "+") + bin.substr(this.bitOffset);
614             }
615 
616             txt += bin + "    0x" + hi16 + " 0x" + lo16 + "    " + this.left_pad(hi, 3, " ") + " " + this.left_pad(lo, 3, " ") + "    " + this.left_pad(chr, 5, " ") + "    " + String.fromCharCode(chr) + "\n";
617         }
618 
619         if (showLastAdd) {
620             strBit = (this.data.length * 16 - ((16 - this.lastCharBits) % 16) - (this.lastBitsAdded % 16)) % 16;
621             endBit = strBit + this.lastBitsAdded;
622 
623             for (f = 0; f <= endBit - 1; f++) {
624                 if (f === 16) {
625                     txt += "\n";
626                 }
627 
628                 if (f < strBit) {
629                     txt += " ";
630                 } else if (f === strBit) {
631                     if (this.lastBitsAdded === 1) {
632                         txt += "*";
633                     } else {
634                         txt += "<";
635                     }
636                 } else if (f === endBit - 1) {
637                     txt += ">";
638                 } else {
639                     txt += "-";
640                 }
641             }
642         }
643 
644         txt += "\nSIZE: " + this.size() + " bits / " + this.data.length + " characters\n";
645 
646         return txt;
647     };
648 
649     /**
650      * Left pads a string.
651      * @param val Value to be padded
652      * @param len Length to be padded to.
653      * @param str String to pad with.
654      * @return {String} The padded string.
655      */
656     JSBitStream.prototype.left_pad = function (val, len, str) {
657         var w;
658         val += "";
659         if (str === undefined) { str = "0"; }
660         for (w = val.length; w < len; w++) { val = str.charAt(0) + val; }
661         return val;
662     };
663 
664     /**
665      * Log wrapper.
666      * @param severity {Number} Severity of the log message (0 - log, 1 - warn, 2 - error)
667      * @param message {String} The message to be logged.
668      */
669     JSBitStream.prototype.log = function (severity, message) {
670         switch (severity) {
671         case 2:
672             if (console.error !== undefined) {
673                 console.error(message);
674             } else {
675                 this.log(0, "ERROR: " + message);
676             }
677             break;
678         case 1:
679             if (console.warn !== undefined) {
680                 console.warn(message);
681             } else {
682                 this.log(0, "WARNING: " + message);
683             }
684             break;
685         default: // 0
686             console.log(message);
687         }
688     };
